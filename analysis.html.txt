<script>
window.onload=()=>{
    const m=localStorage.getItem('babyFoodRecordMode')||'light-mode';
    document.documentElement.className=m;
    document.getElementById('modeSwitch').innerText=m=='light-mode'?'🌙 夜间':'☀️ 日间';
    renderTable();
}
document.getElementById('modeSwitch').onclick=()=>{
    const root=document.documentElement;
    const now=root.className;
    const next=now=='light-mode'?'dark-mode':'light-mode';
    root.className=next;
    localStorage.setItem('babyFoodRecordMode',next);
    document.getElementById('modeSwitch').innerText=next=='light-mode'?'🌙 夜间':'☀️ 日间';
}
function renderTable(){
    const arr=JSON.parse(localStorage.getItem('babyFoodRecords'))||[];
    const b=document.getElementById('tableBody');
    if(arr.length===0){b.innerHTML='<tr><td colspan="4" style="text-align:center">暂无记录</td></tr>';return}
    arr.sort((a,b)=>new Date(b.date+b.time)-new Date(a.date+a.time));
    b.innerHTML=arr.map(i=>`
    <tr>
        <td>${i.date} ${i.time}</td>
        <td>${i.food}</td>
        <td>${i.amount}</td>
        <td>${i.babyStatus=='happy'?'😊 佳':'😢 一般'}</td>
    </tr>`).join('');
    const happy=arr.filter(i=>i.babyStatus=='happy').length;
    const rate=arr.length?((happy/arr.length)*100).toFixed(1):0;
    document.getElementById('summary').innerText=`共 ${arr.length} 条 | 好评率 ${rate}%`;
}
</script>
